birds.js

let birds = JSON.parse(localStorage.getItem('budgibook_birds') || '[]');

function saveBirds() {
  localStorage.setItem('budgibook_birds', JSON.stringify(birds));
}

function renderBirds() {
  const list = document.getElementById('birdList');

  if (!birds.length) {
    list.innerHTML = `<p>Aucun oiseau enregistré pour le moment.</p>`;
    return;
  }

  list.innerHTML = birds.map((bird, index) => `
    <div class="bird-card">
      <h3>${bird.name || 'Sans nom'}</h3>
      <p><strong>Bague :</strong> ${bird.ring || 'Non renseignée'}</p>
      <p><strong>Sexe :</strong> ${bird.sex || 'Non renseigné'}</p>
      <p><strong>Mutation :</strong> ${bird.mutation || 'Non renseignée'}</p>
      <p><strong>Naissance :</strong> ${bird.birth || 'Non renseignée'}</p>
      <p><strong>Statut :</strong> ${bird.status || 'Non renseigné'}</p>
      <p><strong>Notes :</strong> ${bird.notes || 'Aucune note'}</p>
      <button onclick="deleteBird(${index})">Supprimer</button>
    </div>
  `).join('');
}

function deleteBird(index) {
  birds.splice(index, 1);
  saveBirds();
  renderBirds();
}

renderBirds();
